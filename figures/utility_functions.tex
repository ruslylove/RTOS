% axes + deadline line, drawn once per panel
\newcommand{\panelaxes}{%
  \draw[axgray,line width=0.8pt,-Stealth] (0,-1.65) -- (0,2.05);
  \draw[axgray,line width=0.8pt,-Stealth] (0,0) -- (4.4,0);
  \node[axgray,font=\scriptsize,above] at (0,2.05) {value};
  \node[axgray,font=\scriptsize,below] at (4.3,0) {time};
  \draw[rtred,line width=0.8pt,dashed] (2,-1.5) -- (2,1.95);
  \node[rtred,font=\tiny,below] at (2,-1.5) {deadline};
}

\begin{document}
\begin{tikzpicture}[font=\sffamily,x=1cm,y=1cm]

      (0,1.05) -- (2,1.05) -- (2,-1.3) -- (4,-1.3);

  % ===== Panel 2 — Firm =====

  \end{scope}

\end{tikzpicture}
